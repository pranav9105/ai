def is_safe(board, row, col):
    for r in range(row):
        if board[r] == col:
            return False
        if abs(board[r] - col) == abs(r - row):
            return False
    return True

def solve(board, row, n, sols):
    if row == n:
        sols.append(board[:])
        return
    for col in range(n):
        if is_safe(board, row, col):
            board[row] = col
            solve(board, row + 1, n, sols)
            board[row] = -1

n = 4
sols = []
solve([-1] * n, 0, n, sols)
print("Total solutions:", len(sols))

    print(s)
